**Goal:** Upgrade the telnet frontend to a three-region split-screen layout: room display pinned at top (rows 1–6), scrolling console in the middle (VT100 scroll region), prompt+input pinned at bottom — with graceful degradation to the existing scrolling-stream mode for clients that don't negotiate NAWS.

**Architecture:** `Conn` gains `Width`, `Height`, `inputBuf`, and `splitScreen` fields. `Negotiate()` is extended to send `IAC DO NAWS`; `handleIAC()` SB case is extended to parse the NAWS subnegotiation. New screen control methods (`InitScreen`, `WriteRoom`, `WriteConsole`, `WritePromptSplit`) handle all cursor/region logic. `forwardServerEvents` and `commandLoop` in `game_bridge.go` are updated to call the new split-screen methods when `conn.splitScreen` is true, leaving legacy mode untouched otherwise.

**Tech Stack:** Go, `net.Pipe` in tests, `pgregory.net/rapid` for property tests, VT100/ANSI escape sequences.

case SB:
    // Sub-negotiation: read until IAC SE, capturing data bytes.
    var subdata []byte
    var opt byte
    first := true
    for {
        b, err := c.reader.ReadByte()
        if err != nil {
            return err
        }
        if first {
            opt = b
            first = false
            continue
        }
        if b == IAC {
            next, err := c.reader.ReadByte()
            if err != nil {
                return err
            }
            if next == SE {
                break
            }
            // IAC IAC inside SB = literal 0xFF
            subdata = append(subdata, 0xFF)
            continue
        }
        subdata = append(subdata, b)
    }
    if opt == OptNAWS && len(subdata) == 4 {
        w := int(subdata[0])<<8 | int(subdata[1])
        h := int(subdata[2])<<8 | int(subdata[3])
        c.mu.Lock()
        c.width = w
        c.height = h
        c.mu.Unlock()
    }
```

// tryReadEscapeSeq attempts to read a VT100 CSI escape sequence after ESC has been consumed.
// Returns the sentinel string if recognized, or "" to indicate an ignored sequence.
// All bytes of the sequence (including the final byte) are consumed from the reader.
func (c *Conn) tryReadEscapeSeq() (sentinel string, consumed bool) {
    // Peek at next byte
    next, err := c.reader.ReadByte()
    if err != nil || next != '[' {
        // Not a CSI sequence; discard the single byte already read
        return "", true
    }
    // Read the final byte of the CSI sequence
    final, err := c.reader.ReadByte()
    if err != nil {
        return "", true
    }
    switch final {
    case 'A':
        return "\x00UP", true
    case 'B':
        return "\x00DOWN", true
    default:
        return "", true
    }
}
```

package telnet

import (
    "fmt"
    "strings"
)

const roomRegionRows = 6

// InitScreen initializes the split-screen layout for the connection.
// It hides the cursor, clears the screen, sets the VT100 scroll region,
// draws dividers, and shows the cursor.
//
// Precondition: conn.splitScreen must be true; conn.width and conn.height must be > 0.
// Postcondition: Terminal is configured for three-region layout.
func (c *Conn) InitScreen() error {
    c.mu.Lock()
    w := c.width
    h := c.height
    c.mu.Unlock()

    divider := strings.Repeat("═", w)

    var buf strings.Builder
    // Hide cursor
    buf.WriteString("\033[?25l")
    // Clear screen and home
    buf.WriteString("\033[2J\033[H")
    // Set scroll region: rows 8..(H-2)
    fmt.Fprintf(&buf, "\033[8;%dr", h-2)
    // Draw top divider at row 7
    fmt.Fprintf(&buf, "\033[7;1H%s", divider)
    // Draw bottom divider at row H-1
    fmt.Fprintf(&buf, "\033[%d;1H%s", h-1, divider)
    // Move cursor to input row
    fmt.Fprintf(&buf, "\033[%d;1H", h)
    // Show cursor
    buf.WriteString("\033[?25h")

    return c.writeRaw(buf.String())
}

// WriteRoom renders content into the pinned room region (rows 1–6).
// Content is split on newlines; only the first 6 lines are displayed.
// Cursor position is saved before and restored after.
//
// Precondition: conn.splitScreen must be true.
// Postcondition: Rows 1–6 contain the room content; cursor is at its prior position.
func (c *Conn) WriteRoom(content string) error {
    c.mu.Lock()
    w := c.width
    c.mu.Unlock()

    lines := strings.Split(content, "\n")

    var buf strings.Builder
    // Save cursor
    buf.WriteString("\033[s")
    // Move to row 1
    buf.WriteString("\033[1;1H")

    for row := 0; row < roomRegionRows; row++ {
        // Clear the line
        buf.WriteString("\033[2K")
        if row < len(lines) {
            line := lines[row]
            // Truncate to terminal width (strip ANSI for length, but write raw)
            if len(line) > w {
                line = line[:w]
            }
            buf.WriteString(line)
        }
        if row < roomRegionRows-1 {
            buf.WriteString("\r\n")
        }
    }

    // Restore cursor
    buf.WriteString("\033[u")

    return c.writeRaw(buf.String())
}

// WriteConsole writes a message into the VT100 scroll region and redraws
// the bottom divider and prompt. The player's partially-typed input is preserved.
//
// Precondition: conn.splitScreen must be true; conn.height must be > 0.
// Postcondition: Message appears in console; prompt and inputBuf are redrawn at row H.
func (c *Conn) WriteConsole(text string) error {
    c.mu.Lock()
    h := c.height
    w := c.width
    input := c.inputBuf
    c.mu.Unlock()

    divider := strings.Repeat("═", w)

    var buf strings.Builder
    // Save cursor, move to bottom of scroll region, scroll up one line, write text
    fmt.Fprintf(&buf, "\033[%d;1H\r\n", h-2)
    // Wrap text at width and write
    for _, line := range wrapText(text, w) {
        buf.WriteString(line)
        buf.WriteString("\r\n")
    }
    // Redraw bottom divider
    fmt.Fprintf(&buf, "\033[%d;1H%s", h-1, divider)
    // Redraw input row
    fmt.Fprintf(&buf, "\033[%d;1H\033[2K", h)
    // Write preserved input
    if input != "" {
        buf.WriteString(input)
    }

    return c.writeRaw(buf.String())
}

// WritePromptSplit writes the prompt and any buffered input at the input row (row H).
//
// Precondition: conn.splitScreen must be true; conn.height must be > 0.
// Postcondition: Prompt appears at row H, cursor is after the prompt+input.
func (c *Conn) WritePromptSplit(prompt string) error {
    c.mu.Lock()
    h := c.height
    input := c.inputBuf
    c.mu.Unlock()

    var buf strings.Builder
    fmt.Fprintf(&buf, "\033[%d;1H\033[2K%s%s", h, prompt, input)

    return c.writeRaw(buf.String())
}

// SetInputBuf updates the buffered player input for redraw after server events.
func (c *Conn) SetInputBuf(s string) {
    c.mu.Lock()
    defer c.mu.Unlock()
    c.inputBuf = s
}

// EnableSplitScreen marks the connection as operating in split-screen mode.
// Called after NAWS dimensions are known and InitScreen has been sent.
func (c *Conn) EnableSplitScreen() {
    c.mu.Lock()
    defer c.mu.Unlock()
    c.splitScreen = true
}

// writeRaw writes a raw string to the connection (bypassing the line-oriented helpers).
func (c *Conn) writeRaw(s string) error {
    c.mu.Lock()
    defer c.mu.Unlock()
    if c.writeTimeout > 0 {
        _ = c.raw.SetWriteDeadline(timeNow().Add(c.writeTimeout))
    }
    _, err := c.raw.Write([]byte(s))
    return err
}

// wrapText splits text into lines of at most width runes.
// Lines already shorter than width are returned as-is.
func wrapText(text string, width int) []string {
    if width <= 0 {
        return []string{text}
    }
    var result []string
    for _, line := range strings.Split(text, "\n") {
        for len(line) > width {
            result = append(result, line[:width])
            line = line[width:]
        }
        result = append(result, line)
    }
    return result
}

## Task 5: NAWS timeout and `EnableSplitScreen` in `acceptor.go` / `gameBridge`

**Files:**
- Modify: `internal/frontend/telnet/acceptor.go`
- Modify: `internal/frontend/handlers/game_bridge.go`
- Modify: `internal/frontend/handlers/game_bridge_test.go` (if it exists; create if not)

This task wires the NAWS wait into the connect flow. After `Negotiate()` returns, the acceptor waits up to 1 second for NAWS. If dimensions arrive and are non-zero, `EnableSplitScreen()` is called; otherwise the connection runs in legacy mode.

Because `ReadLine` is the only input path, NAWS arrives as a side-effect of the first `ReadLine` call. The simplest correct approach is:

1. After `Negotiate()`, set a 1-second read deadline.
2. Call `conn.ReadLine()` in a loop until either: (a) dimensions are non-zero (NAWS arrived), (b) 1 second has elapsed, or (c) the client sends a non-empty first line.
3. If dimensions are non-zero, call `conn.InitScreen()` and `conn.EnableSplitScreen()`.
4. Then hand off to `HandleSession`.

However, `acceptor.go` calls `handler.HandleSession(ctx, conn)` and doesn't own the first read. A cleaner approach: add `conn.AwaitNAWS(timeout)` that reads (and discards) bytes for up to `timeout` or until NAWS arrives.

### Step 3: Implement `AwaitNAWS` in `conn.go`

```go
// AwaitNAWS waits up to timeout for the client to send a NAWS subnegotiation.
// It reads and discards all incoming IAC sequences without building a line buffer.
// Returns true if non-zero dimensions were received before the timeout.
//
// Precondition: Negotiate() must already have been called (so IAC DO NAWS was sent).
// Postcondition: conn.width and conn.height are set if NAWS was received.
func (c *Conn) AwaitNAWS(timeout time.Duration) bool {
    deadline := time.Now().Add(timeout)
    _ = c.raw.SetReadDeadline(deadline)
    defer func() { _ = c.raw.SetReadDeadline(time.Time{}) }()

    for {
        b, err := c.reader.ReadByte()
        if err != nil {
            return false
        }
        if b != IAC {
            continue
        }
        if err := c.handleIAC(); err != nil {
            return false
        }
        w, h := c.Dimensions()
        if w > 0 && h > 0 {
            return true
        }
    }
}
```
