**Step 1: Add `checkReactiveStrikes` to `round.go`.**

Add after `ResolveRound`:
```go
// checkReactiveStrikes fires a free ResolveAttack from any NPC combatant that was
// within melee range of moverID before the move and is now outside melee range.
//
// Precondition: moverID must be a valid combatant ID in cbt; oldPos and newPos are in feet.
// Postcondition: Returns RoundEvents for each reactive strike; combat HP updated via targetUpdater.
func checkReactiveStrikes(cbt *Combat, moverID string, oldPos, newPos int, src Source, targetUpdater func(id string, hp int)) []RoundEvent {
    mover := findCombatantByID(cbt, moverID)
    if mover == nil {
        return nil
    }
    var events []RoundEvent
    for _, npc := range cbt.Combatants {
        if npc.Kind != KindNPC || npc.IsDead() {
            continue
        }
        wasInMelee := abs(npc.Position-oldPos) <= 5
        isOutOfMelee := abs(npc.Position-newPos) > 5
        if !wasInMelee || !isOutOfMelee {
            continue
        }
        // NPC gets a free ResolveAttack as a Reactive Strike.
        result := ResolveAttack(npc, mover, src)
        if result.Outcome == CritSuccess || result.Outcome == Success {
            dmg := result.EffectiveDamage()
            mover.ApplyDamage(dmg)
            targetUpdater(mover.ID, mover.CurrentHP)
        }
        narrative := attackNarrative(result, npc.Name, mover.Name, result.WeaponName) +
            fmt.Sprintf(" (Reactive Strike as %s moves away)", mover.Name)
        events = append(events, RoundEvent{
            AttackResult: &result,
            ActionType:   ActionAttack,
            ActorID:      npc.ID,
            ActorName:    npc.Name,
            Narrative:    narrative,
        })
    }
    return events
}

func abs(x int) int {
    if x < 0 {
        return -x
    }
    return x
}

// handleShove performs an athletics check vs the target's Level+10 DC (Fortitude placeholder).
// On success, target is pushed 5ft away; on critical success, pushed 10ft. Costs 1 AP.
//
// Precondition: uid in active combat; target NPC in room.
// Postcondition: On success, target Position increased by 5 or 10.
func (s *GameServiceServer) handleShove(uid string, req *gamev1.ShoveRequest) (*gamev1.ServerEvent, error) {
    sess, ok := s.sessions.GetPlayer(uid)
    if !ok {
        return nil, fmt.Errorf("player %q not found", uid)
    }
    if sess.Status != statusInCombat {
        return errorEvent("Shove is only available in combat."), nil
    }
    if req.GetTarget() == "" {
        return errorEvent("Usage: shove <target>"), nil
    }
    inst := s.npcMgr.FindInRoom(sess.RoomID, req.GetTarget())
    if inst == nil {
        return errorEvent(fmt.Sprintf("Target %q not found in current room.", req.GetTarget())), nil
    }
    if err := s.combatH.SpendAP(uid, 1); err != nil {
        return errorEvent(err.Error()), nil
    }
    rollResult, err := s.dice.RollExpr("1d20")
    if err != nil {
        return nil, fmt.Errorf("handleShove: rolling d20: %w", err)
    }
    roll := rollResult.Total()
    bonus := skillRankBonus(sess.Skills["athletics"])
    total := roll + bonus
    dc := inst.Level + 10
    detail := fmt.Sprintf("Shove (athletics DC %d): rolled %d+%d=%d", dc, roll, bonus, total)
    if total < dc {
        return messageEvent(detail + " — failure. Your shove attempt fails."), nil
    }
    // Find NPC combatant and push it.
    push := 5
    if total >= dc+10 {
        push = 10
    }
    cbt, ok := s.combatH.GetCombatForRoom(sess.RoomID)
    if ok {
        npcCbt := cbt.GetCombatant(inst.ID)
        if npcCbt != nil {
            npcCbt.Position += push
        }
    }
    return messageEvent(fmt.Sprintf("%s pushed %dft away.", inst.Name(), push) + " " + detail + " — success!"), nil
}
```

// handleStep moves the player 5ft toward or away, without triggering Reactive Strike.
// Combat only; costs 1 AP.
//
// Precondition: uid in active combat.
// Postcondition: Player Position updated ±5; no Reactive Strike fires.
func (s *GameServiceServer) handleStep(uid string, req *gamev1.StepRequest) (*gamev1.ServerEvent, error) {
    sess, ok := s.sessions.GetPlayer(uid)
    if !ok {
        return nil, fmt.Errorf("player %q not found", uid)
    }
    if sess.Status != statusInCombat {
        return errorEvent("Step is only available in combat."), nil
    }
    if err := s.combatH.SpendAP(uid, 1); err != nil {
        return errorEvent(err.Error()), nil
    }
    cbt, ok := s.combatH.GetCombatForRoom(sess.RoomID)
    if !ok {
        return errorEvent("No active combat found."), nil
    }
    combatant := cbt.GetCombatant(uid)
    if combatant == nil {
        return errorEvent("You are not a combatant in this fight."), nil
    }
    dir := req.GetDirection()
    switch dir {
    case "away":
        combatant.Position += 5
    case "toward", "":
        dir = "toward"
        if combatant.Position-5 < 0 {
            combatant.Position = 0
        } else {
            combatant.Position -= 5
        }
    default:
        return errorEvent(fmt.Sprintf("Unknown direction %q. Use 'toward' or 'away'.", dir)), nil
    }
    // NOTE: Step explicitly does NOT trigger Reactive Strike.
    return messageEvent(fmt.Sprintf("You step %s carefully.", dir)), nil
}

```go
// handleTumble attempts to move through a target's space.
// Success: player moves 5ft toward target, no Reactive Strike.
// Failure: no move, Reactive Strike from target fires.
// Combat only; costs 1 AP.
//
// Precondition: uid in active combat; target NPC in room.
// Postcondition: On success, player Position updated 5ft; on failure, Reactive Strike events broadcast.
func (s *GameServiceServer) handleTumble(uid string, req *gamev1.TumbleRequest) (*gamev1.ServerEvent, error) {
    sess, ok := s.sessions.GetPlayer(uid)
    if !ok {
        return nil, fmt.Errorf("player %q not found", uid)
    }
    if sess.Status != statusInCombat {
        return errorEvent("Tumble is only available in combat."), nil
    }
    if req.GetTarget() == "" {
        return errorEvent("Usage: tumble <target>"), nil
    }
    inst := s.npcMgr.FindInRoom(sess.RoomID, req.GetTarget())
    if inst == nil {
        return errorEvent(fmt.Sprintf("Target %q not found in current room.", req.GetTarget())), nil
    }
    if err := s.combatH.SpendAP(uid, 1); err != nil {
        return errorEvent(err.Error()), nil
    }
    rollResult, err := s.dice.RollExpr("1d20")
    if err != nil {
        return nil, fmt.Errorf("handleTumble: rolling d20: %w", err)
    }
    roll := rollResult.Total()
    bonus := skillRankBonus(sess.Skills["acrobatics"])
    total := roll + bonus
    dc := inst.Level + 10
    detail := fmt.Sprintf("Tumble Through (acrobatics DC %d): rolled %d+%d=%d", dc, roll, bonus, total)

    cbt, hasCbt := s.combatH.GetCombatForRoom(sess.RoomID)

    if total < dc {
        // Failure: no position change; Reactive Strike from target fires.
        if hasCbt {
            playerCbt := cbt.GetCombatant(uid)
            npcCbt := cbt.GetCombatant(inst.ID)
            if playerCbt != nil && npcCbt != nil {
                rsEvents := combat.CheckReactiveStrikeFrom(cbt, npcCbt, playerCbt, s.dice, func(id string, hp int) {
                    s.combatH.UpdateCombatantHP(id, hp)
                })
                for _, evt := range rsEvents {
                    protoEvt := s.worldH.roundEventToProto(evt)
                    if protoEvt != nil {
                        s.broadcastCombatEvent(sess.RoomID, uid, protoEvt)
                    }
                }
            }
        }
        return messageEvent(detail + " — failure. Blocked! " + inst.Name() + " strikes as you attempt to pass."), nil
    }

    // Success: move 5ft toward target, no Reactive Strike.
    if hasCbt {
        playerCbt := cbt.GetCombatant(uid)
        npcCbt := cbt.GetCombatant(inst.ID)
        if playerCbt != nil && npcCbt != nil {
            // Move toward target (positive direction).
            if npcCbt.Position > playerCbt.Position {
                playerCbt.Position += 5
            } else {
                if playerCbt.Position-5 < 0 {
                    playerCbt.Position = 0
                } else {
                    playerCbt.Position -= 5
                }
            }
        }
    }
    return messageEvent(detail + fmt.Sprintf(" — success! You tumble through %s's space.", inst.Name())), nil
}
```

// CheckReactiveStrikeFrom fires a single Reactive Strike from attacker against target.
//
// Precondition: attacker and target must be non-nil combatants in cbt.
// Postcondition: Returns []RoundEvent; HP updated via targetUpdater.
func CheckReactiveStrikeFrom(cbt *Combat, attacker, target *Combatant, src Source, targetUpdater func(id string, hp int)) []RoundEvent {
    result := ResolveAttack(attacker, target, src)
    if result.Outcome == CritSuccess || result.Outcome == Success {
        dmg := result.EffectiveDamage()
        target.ApplyDamage(dmg)
        targetUpdater(target.ID, target.CurrentHP)
    }
    narrative := attackNarrative(result, attacker.Name, target.Name, result.WeaponName) +
        " (Reactive Strike)"
    return []RoundEvent{{
        AttackResult: &result,
        ActionType:   ActionAttack,
        ActorID:      attacker.ID,
        ActorName:    attacker.Name,
        Narrative:    narrative,
    }}
}
