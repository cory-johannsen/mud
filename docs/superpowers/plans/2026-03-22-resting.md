# Resting Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Extend the `rest` command with location-aware routing: motel rest (instant, safe room with motel NPC) and camping rest (timed, non-safe zone, requires gear), including partial restoration on camping interruption. Also reduce passive HP regeneration from 30-second to 5-minute ticks so that resting is the practical recovery method after combat.

**Architecture:** `handleRest` gains a pre-check sequence before invoking the existing long-rest logic. Camping state (`CampingActive`, `CampingStartTime`, `CampingDuration`) is added to `PlayerSession`. The game clock tick calls `checkCampingStatus` to detect completion or enemy presence. `handleMove` cancels camping with partial restoration if the player moves. Motel rest requires the motel NPC type from the `non-combat-npcs` feature — the routing check is implemented now but the motel NPC itself is deferred. `RegenInterval` in `regen.go` is changed from 30 seconds to 5 minutes.

**Tech Stack:** Go, testify, rapid

**Dependencies:**
- `room-danger-levels` (already done) — `danger.EffectiveDangerLevel()` used to determine safe vs. non-safe
- `exploration` (already planned) — `sess.ExploreMode != ""` for camping eligibility check
- `non-combat-npcs` (blocked) — motel NPC type definition; motel path compiles but returns "no motel available" until motel NPCs exist in content

```go
func (s *GameServiceServer) handleRest(uid string) error {
    sess, ok := s.sessions.GetPlayer(uid)
    if !ok {
        return fmt.Errorf("session not found")
    }

    // REQ-REST-1: block in combat (existing guard — keep as-is)
    if sess.Status == statusInCombat {
        return s.sendError(uid, "You cannot rest during combat.")
    }

    // Already camping — inform the player
    if sess.CampingActive {
        elapsed := time.Since(sess.CampingStartTime)
        remaining := sess.CampingDuration - elapsed
        if remaining < 0 {
            remaining = 0
        }
        return s.sendConsole(uid, fmt.Sprintf("You are already camping. Time remaining: %s.", remaining.Round(time.Second)))
    }

    // Determine room safety
    room, ok := s.world.GetRoom(sess.RoomID)
    if !ok {
        return s.sendError(uid, "Cannot determine your location.")
    }
    zone, zoneOK := s.world.GetZone(room.ZoneID)
    zoneDangerLevel := ""
    if zoneOK {
        zoneDangerLevel = zone.DangerLevel
    }
    effectiveDanger := danger.EffectiveDangerLevel(zoneDangerLevel, room.DangerLevel)
    isSafe := effectiveDanger == danger.Safe

    // REQ-REST-2: safe room + motel NPC → motel rest (deferred: non-combat-npcs not yet implemented)
    if isSafe {
        // Check for motel NPC in room (motel NPC type = "motel" per non-combat-npcs spec)
        npcsInRoom := s.npcMgr.InstancesInRoom(sess.RoomID)
        for _, npc := range npcsInRoom {
            if npc.Type == "motel" {
                return s.handleMotelRest(uid, sess, npc)
            }
        }
        // REQ-REST-4: safe room but no motel NPC
        return s.sendError(uid, "There is no motel here to rest at.")
    }

    // REQ-REST-3: non-safe zone + active exploration mode → camping rest
    if sess.ExploreMode != "" {
        return s.startCamping(uid, sess)
    }

    // REQ-REST-4: neither condition met
    return s.sendError(uid, "You cannot rest here. You must be in exploration mode to camp in a dangerous area.")
}
```

```go
// startCamping validates gear requirements and starts a camping rest session.
func (s *GameServiceServer) startCamping(uid string, sess *session.PlayerSession) error {
    // REQ-REST-10: require sleeping_bag (exact item ID)
    hasSleepingBag := len(sess.Backpack.FindByItemDefID("sleeping_bag")) > 0

    // REQ-REST-11: require at least one fire_material-tagged item
    hasFireMaterial := false
    campingGearCount := 0
    for _, item := range sess.Backpack.Items() {
        def, ok := s.invRegistry.Item(item.ItemDefID)
        if !ok {
            continue
        }
        if def.HasTag("fire_material") {
            hasFireMaterial = true
        }
        if def.HasTag("camping_gear") {
            campingGearCount += item.Quantity
        }
    }


    // REQ-REST-12: collect all missing-gear errors
    var missing []string
    if !hasSleepingBag {
        missing = append(missing, "a sleeping bag (sleeping_bag)")
    }
    if !hasFireMaterial {
        missing = append(missing, "fire-starting materials (fire_material)")
    }
    if len(missing) > 0 {
        return s.sendError(uid, fmt.Sprintf("You need the following to camp: %s.", strings.Join(missing, "; ")))
    }

    // REQ-REST-13/14: compute duration
    base := 5 * time.Minute
    reduction := time.Duration(campingGearCount) * 30 * time.Second
    duration := base - reduction
    if duration < 2*time.Minute {
        duration = 2 * time.Minute // REQ-REST-14 minimum
    }

    // REQ-REST-17: gear is NOT consumed
    sess.CampingActive = true
    sess.CampingStartTime = time.Now()
    sess.CampingDuration = duration

    return s.sendConsole(uid, fmt.Sprintf(
        "You set up camp. You will rest for %s. Use 'move' to cancel early.",
        duration.Round(time.Second)))
}
```

```go
// cancelCamping applies partial restoration and clears camping state.
// Called on enemy room entry or player move while camping.
func (s *GameServiceServer) cancelCamping(uid string, sess *session.PlayerSession, reason string) {
    elapsed := time.Since(sess.CampingStartTime)
    duration := sess.CampingDuration
    if duration == 0 {
        duration = 1 // avoid divide-by-zero
    }

    sess.CampingActive = false
    sess.CampingStartTime = time.Time{}
    sess.CampingDuration = 0

    // REQ-REST-15/16: partial restoration = floor(elapsed/duration * max)
    frac := float64(elapsed) / float64(duration)
    if frac > 1.0 {
        frac = 1.0
    }

    // Partial HP restore
    hpGain := int(math.Floor(frac * float64(sess.MaxHP-sess.CurrentHP)))
    if hpGain > 0 {
        sess.CurrentHP = min(sess.CurrentHP+hpGain, sess.MaxHP)
        _ = s.charSaver.SaveState(context.Background(), sess.CharacterID, sess.RoomID, sess.CurrentHP)
    }

    // REQ-REST-15/16: partial tech pool restore
    // RestorePartial is added to the repo in this plan (see Task 3 Step 5b).
    _ = s.spontaneousUsePoolRepo.RestorePartial(context.Background(), sess.CharacterID, frac)
    // REQ-REST-17: gear is NOT consumed

    s.sendConsole(uid, fmt.Sprintf("Camping interrupted: %s. Partial restoration applied (%.0f%% complete).",
        reason, frac*100))
}
```

```go
// checkCampingStatus is called on each game clock tick.
// Completes camping if elapsed, or cancels if enemies are present (REQ-REST-15).
func (s *GameServiceServer) checkCampingStatus(uid string) {
    sess, ok := s.sessions.GetPlayer(uid)
    if !ok || !sess.CampingActive {
        return
    }

    // REQ-REST-15: check for enemy presence (hostile NPCs in room)
    npcsInRoom := s.npcMgr.InstancesInRoom(sess.RoomID)
    for _, npc := range npcsInRoom {
        if npc.Disposition == "hostile" { // hostile NPCs interrupt camping (REQ-REST-15)
            s.cancelCamping(uid, sess, "enemies spotted")
            return
        }
    }

    // REQ-REST-18: check completion
    if time.Since(sess.CampingStartTime) >= sess.CampingDuration {
        sess.CampingActive = false
        sess.CampingStartTime = time.Time{}
        sess.CampingDuration = 0
        s.sendConsole(uid, "You wake up refreshed.")
        _ = s.applyFullLongRest(uid, sess)
    }
}
```
