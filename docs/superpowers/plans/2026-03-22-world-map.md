```go
    if !ok {
        return
```

```go
            }
        }
```

```go
package gameserver

import (
	"fmt"

	gamev1 "github.com/cory-johannsen/mud/internal/gameserver/gamev1"
)

// handleTravel fast-travels the player to the named zone's start room.
//
// Preconditions (checked in order):
//  1. req.ZoneId must exist in the world model.
//  2. Player must have discovered the target zone (AutomapCache entry non-empty).
//  3. Player must not be in combat.
//  4. Player must have no non-zero WantedLevel entries.
//  5. Player must not already be in the target zone.
//
// Postcondition: On success, sess.RoomID is set to targetZone.StartRoom and a
// confirmation MessageEvent is returned followed by a RoomView event via the
// normal room-entry path.
func (s *GameServiceServer) handleTravel(uid string, req *gamev1.TravelRequest) (*gamev1.ServerEvent, error) {
	sess, ok := s.sessions.GetPlayer(uid)
	if !ok {
		return nil, fmt.Errorf("player %q not found", uid)
	}

	// Precondition 1: zone must exist.
	targetZone, ok := s.world.GetZone(req.GetZoneId())
	if !ok {
		return messageEvent("That zone does not exist."), nil
	}

	// Precondition 2: zone must be discovered.
	if len(sess.AutomapCache[req.GetZoneId()]) == 0 {
		return messageEvent("You don't know how to get there."), nil
	}

	// Precondition 3: no active combat.
	if sess.Status == statusInCombat {
		return messageEvent("You can't travel while in combat."), nil
	}

	// Precondition 4: no wanted levels.
	for _, level := range sess.WantedLevel {
		if level != 0 {
			return messageEvent("You can't travel while Wanted."), nil
		}
	}

	// Precondition 5: not already in target zone.
	currentRoom, ok := s.world.GetRoom(sess.RoomID)
	if ok && currentRoom.ZoneID == req.GetZoneId() {
		return messageEvent("You're already there."), nil
	}

	// Relocate to target zone's start room.
	startRoom, ok := s.world.GetRoom(targetZone.StartRoom)
	if !ok {
		return messageEvent("That zone is unreachable right now."), nil
	}
	sess.RoomID = startRoom.ID

	// Fire normal room-entry hooks.
	if s.onRoomEntry != nil {
		s.onRoomEntry(uid, startRoom.ID)
	}

	// Build confirmation message.
	travelMsg := fmt.Sprintf("You make your way to %s.", targetZone.Name)
	return messageEvent(travelMsg), nil
}
```

```go
// RenderWorldMap formats a world map from MapResponse.WorldTiles into a colored
// terminal string using the same grid-cell layout as RenderMap.
//
// Precondition: resp must be non-nil; width > 0.
// Postcondition: Returns a multi-line ANSI-colored string with legend.
// Layout: two-column (grid left, legend right) at width >= 100; single-column otherwise.
func RenderWorldMap(resp *gamev1.MapResponse, width int) string {
	tiles := resp.GetWorldTiles()
	if len(tiles) == 0 {
		return telnet.Colorize(telnet.BrightYellow, "World Map") + "\r\n" +
			telnet.Colorize(telnet.Dim, "(no zones discovered)") + "\r\n"
	}

	const cellStride = 5 // 4-char cell + 1-char connector slot
	const rowStride = 2  // cell row + connector row

	// Find coordinate bounds for normalization.
	minX, minY := tiles[0].WorldX, tiles[0].WorldY
	for _, t := range tiles {
		if t.WorldX < minX {
			minX = t.WorldX
		}
		if t.WorldY < minY {
			minY = t.WorldY
		}
	}

	// Sort tiles top-to-bottom, left-to-right for legend number assignment.
	sorted := make([]*gamev1.WorldZoneTile, len(tiles))
	copy(sorted, tiles)
	sort.Slice(sorted, func(i, j int) bool {
		if sorted[i].WorldY != sorted[j].WorldY {
			return sorted[i].WorldY < sorted[j].WorldY
		}
		return sorted[i].WorldX < sorted[j].WorldX
	})

	// Assign legend numbers and build a coord→tile index.
	type tileInfo struct {
		tile   *gamev1.WorldZoneTile
		legend int // 1-based
	}
	byCoord := make(map[[2]int32]*tileInfo, len(sorted))
	legendEntries := make([]string, 0, len(sorted))
	for i, t := range sorted {
		num := i + 1
		key := [2]int32{t.WorldX, t.WorldY}
		byCoord[key] = &tileInfo{tile: t, legend: num}
		label := t.ZoneName
		if !t.Discovered {
			label = "???"
		}
		legendEntries = append(legendEntries, fmt.Sprintf("%02d: %s", num, label))
	}

	// Compute grid dimensions.
	maxCol, maxRow := int32(0), int32(0)
	for _, t := range tiles {
		col := (t.WorldX - minX) / 2
		row := (t.WorldY - minY) / 2
		if col > maxCol {
			maxCol = col
		}
		if row > maxRow {
			maxRow = row
		}
	}
	gridCols := int(maxCol) + 1
	gridRows := int(maxRow) + 1

	// Build grid: rendered rows (cell rows + connector rows).
	renderedHeight := gridRows*rowStride - 1 // last row has no connector row below
	if renderedHeight < 1 {
		renderedHeight = 1
	}
	renderedWidth := gridCols*cellStride // each cell is 4 chars wide + 1 connector slot
	if renderedWidth < 4 {
		renderedWidth = 4
	}

	grid := make([][]byte, renderedHeight)
	for i := range grid {
		row := make([]byte, renderedWidth)
		for j := range row {
			row[j] = ' '
		}
		grid[i] = row
	}

	// Place cell bodies and connectors.
	type placedCell struct {
		gridRow, gridCol int
		info             *tileInfo
	}
	var placed []placedCell

	for _, ti := range byCoord {
		col := int((ti.tile.WorldX - minX) / 2)
		row := int((ti.tile.WorldY - minY) / 2)
		cellRow := row * rowStride
		cellCol := col * cellStride

		var cellBody string
		numStr := fmt.Sprintf("%02d", ti.legend)
		if ti.tile.Current {
			cellBody = "<" + numStr + ">"
		} else if ti.tile.Discovered {
			cellBody = "[" + numStr + "]"
		} else {
			cellBody = "[??]"
		}

		if cellRow < len(grid) && cellCol+3 < len(grid[cellRow]) {
			copy(grid[cellRow][cellCol:cellCol+4], []byte(cellBody))
		}

		placed = append(placed, placedCell{gridRow: cellRow, gridCol: cellCol, info: ti})
	}

	// Render east connectors (between cells differing by exactly 2 on X, 0 on Y).
	for _, a := range placed {
		for _, b := range placed {
			if a.info.tile.WorldY != b.info.tile.WorldY {
				continue
			}
			dx := b.info.tile.WorldX - a.info.tile.WorldX
			if dx != 2 {
				continue
			}
			// Connector goes in the slot between a (col) and b (col+1).
			connRow := a.gridRow
			connCol := a.gridCol + 4
			if connRow < len(grid) && connCol < len(grid[connRow]) {
				grid[connRow][connCol] = '-'
			}
		}
	}

	// Render south connectors (between cells differing by exactly 2 on Y, 0 on X).
	for _, a := range placed {
		for _, b := range placed {
			if a.info.tile.WorldX != b.info.tile.WorldX {
				continue
			}
			dy := b.info.tile.WorldY - a.info.tile.WorldY
			if dy != 2 {
				continue
			}
			connRow := a.gridRow + 1
			connCol := a.gridCol + 1 // center of the 4-char cell
			if connRow < len(grid) && connCol < len(grid[connRow]) {
				grid[connRow][connCol] = '|'
			}
		}
	}

	// Color cell bodies in grid.
	// We build the output line-by-line using the raw grid + ANSI overlays.
	// Since ANSI codes break byte-indexing in grid, we render plain first then
	// reconstruct with color using the placed cell list.

	type coloredSpan struct {
		col   int
		text  string // 4-char colored body
	}
	rowSpans := make(map[int][]coloredSpan, len(placed))
	for _, p := range placed {
		var raw, colored string
		numStr := fmt.Sprintf("%02d", p.info.legend)
		if p.info.tile.Current {
			raw = "<" + numStr + ">"
			colored = telnet.Colorize(telnet.BrightWhite, raw)
		} else if p.info.tile.Discovered {
			raw = "[" + numStr + "]"
			colored = DangerColor(p.info.tile.DangerLevel) + raw + ansiReset
		} else {
			raw = "[??]"
			colored = "\033[37m" + raw + ansiReset
		}
		_ = raw
		rowSpans[p.gridRow] = append(rowSpans[p.gridRow], coloredSpan{col: p.gridCol, text: colored})
	}

	var sb strings.Builder
	sb.WriteString(telnet.Colorize(telnet.BrightYellow, "World Map") + "\r\n")

	buildGridLines := func() []string {
		var lines []string
		for ri, rawRow := range grid {
			spans, hasSpans := rowSpans[ri]
			if !hasSpans {
				lines = append(lines, string(rawRow))
				continue
			}
			// Rebuild row with color spans.
			sort.Slice(spans, func(a, b int) bool { return spans[a].col < spans[b].col })
			var rowSB strings.Builder
			pos := 0
			for _, span := range spans {
				if span.col > pos {
					rowSB.Write(rawRow[pos:span.col])
				}
				rowSB.WriteString(span.text)
				pos = span.col + 4
			}
			if pos < len(rawRow) {
				rowSB.Write(rawRow[pos:])
			}
			lines = append(lines, rowSB.String())
		}
		return lines
	}

	gridLines := buildGridLines()

	if width >= 100 {
		// Two-column: grid on left, legend on right.
		maxLegWidth := 0
		for _, e := range legendEntries {
			if len(e) > maxLegWidth {
				maxLegWidth = len(e)
			}
		}
		for i, gl := range gridLines {
			sb.WriteString(gl)
			if i < len(legendEntries) {
				// Pad to a fixed column then append legend entry.
				rawLen := len(strings.ReplaceAll(strings.ReplaceAll(gl, "\033[0m", ""), "\033[", ""))
				_ = rawLen
				sb.WriteString("  ")
				sb.WriteString(telnet.Colorize(telnet.Cyan, legendEntries[i]))
			}
			sb.WriteString("\r\n")
		}
		for i := len(gridLines); i < len(legendEntries); i++ {
			sb.WriteString(telnet.Colorize(telnet.Cyan, legendEntries[i]))
			sb.WriteString("\r\n")
		}
	} else {
		// Single-column: grid then legend.
		for _, gl := range gridLines {
			sb.WriteString(gl)
			sb.WriteString("\r\n")
		}
		sb.WriteString("\r\n")
		for _, e := range legendEntries {
			sb.WriteString(telnet.Colorize(telnet.Cyan, e))
			sb.WriteString("\r\n")
		}
	}

	return sb.String()
}
```

```go
// handleMapModeInput processes a single line of input while mapMode is true.
// It routes reserved keys, zone selection, travel, and exit — never forwarding to
// the normal command dispatcher.
//
// Precondition: mapMode must be true when called.
// Postcondition: updates mapMode, mapView, mapSelectedZone, lastMapResponse in place.
func (h *AuthHandler) handleMapModeInput(
	line string,
	conn *telnet.Conn,
	stream gamev1.GameService_SessionClient,
	mapMode *bool,
	mapView *string,
	mapSelectedZone *string,
	lastMapResponse *atomic.Value,
	buildPrompt func() string,
	requestID *int,
) {
	// ESC byte (bare 0x1B) exits map mode; ESC sequences (ESC + more) are consumed silently.
	if len(line) > 0 && line[0] == 0x1B {
		if len(line) > 1 {
			// Incomplete escape sequence — silently consume.
			return
		}
		h.exitMapMode(conn, mapMode, mapSelectedZone, buildPrompt)
		return
	}

	lower := strings.TrimSpace(strings.ToLower(line))

	// Reserved key aliases are matched first (REQ-WM-43).
	switch lower {
	case "q", "quit", "esc":
		h.exitMapMode(conn, mapMode, mapSelectedZone, buildPrompt)
		return

	case "z", "zone":
		*mapView = "zone"
		*requestID++
		reqID := fmt.Sprintf("req-%d", *requestID)
		resp, err := h.sendMapRequest(stream, reqID, "zone")
		if err != nil {
			return
		}
		lastMapResponse.Store(resp)
		h.redrawMap(conn, resp, *mapView, buildPrompt)
		return

	case "w", "world":
		*mapView = "world"
		*requestID++
		reqID := fmt.Sprintf("req-%d", *requestID)
		resp, err := h.sendMapRequest(stream, reqID, "world")
		if err != nil {
			return
		}
		lastMapResponse.Store(resp)
		h.redrawMap(conn, resp, *mapView, buildPrompt)
		return

	case "t", "travel":
		if *mapSelectedZone == "" {
			_ = conn.WriteConsole("Select a zone first.")
			_ = conn.WritePromptSplit(buildPrompt())
			return
		}
		*requestID++
		reqID := fmt.Sprintf("req-%d", *requestID)
		_ = stream.Send(&gamev1.ClientMessage{
			RequestId: reqID,
			Payload:   &gamev1.ClientMessage_Travel{Travel: &gamev1.TravelRequest{ZoneId: *mapSelectedZone}},
		})
		return
	}

	// In world view, non-reserved input is treated as a zone selector (REQ-WM-46).
	if *mapView == "world" || *mapView == "" {
		resp, _ := lastMapResponse.Load().(*gamev1.MapResponse)
		if resp != nil {
			zoneID, zoneName, dangerLevel, ok := h.resolveZoneSelector(lower, resp.GetWorldTiles())
			if ok {
				*mapSelectedZone = zoneID
				prompt := fmt.Sprintf("[MAP] Selected: %s (%s)  t=travel  q=exit", zoneName, dangerLevel)
				_ = conn.WritePromptSplit(prompt)
				return
			}
		}
	}

	// Unknown command (REQ-WM-51).
	_ = conn.WriteConsole("Unknown map command. Press q to exit.")
	_ = conn.WritePromptSplit(buildPrompt())
}

// exitMapMode sets mapMode=false, clears mapSelectedZone, redraws console, and resets prompt.
func (h *AuthHandler) exitMapMode(conn *telnet.Conn, mapMode *bool, mapSelectedZone *string, buildPrompt func() string) {
	*mapMode = false
	*mapSelectedZone = ""
	_ = conn.RedrawConsole()
	if conn.IsSplitScreen() {
		_ = conn.WritePromptSplit(buildPrompt())
	} else {

	}
}

// redrawMap clears the console region and re-renders the map at current terminal width.
func (h *AuthHandler) redrawMap(conn *telnet.Conn, resp *gamev1.MapResponse, view string, buildPrompt func() string) {
	w, _ := conn.Dimensions()
	var text string

		text = RenderWorldMap(resp, w)
	} else {
		text = RenderMap(resp, w)
	}
	_ = conn.WriteConsole(text)
	_ = conn.WritePromptSplit(buildPrompt())
}

// resolveZoneSelector matches input against world tile zone names and numbers.
// Returns the first match in lexicographic zone ID order.
func (h *AuthHandler) resolveZoneSelector(input string, tiles []*gamev1.WorldZoneTile) (zoneID, zoneName, dangerLevel string, ok bool) {
	var matches []*gamev1.WorldZoneTile
	for _, t := range tiles {
		if strings.HasPrefix(strings.ToLower(t.ZoneName), input) {
			matches = append(matches, t)
		}
	}
	if len(matches) == 0 {
		return "", "", "", false
	}
	sort.Slice(matches, func(i, j int) bool { return matches[i].ZoneId < matches[j].ZoneId })
	m := matches[0]
	return m.ZoneId, m.ZoneName, m.DangerLevel, true
}
```

### `mapModeState` struct scoping

If threading `mapModeState` through `commandLoop` and `forwardServerEvents` signatures causes too many call site changes, an alternative is to make it a struct with a `sync.Mutex` and pass a pointer. The `forwardServerEvents` and `commandLoop` goroutines share access, so the mutex is required for `mapMode`, `mapView`, and `mapSelectedZone`. `lastMapResponse` is already an `atomic.Value`.

### `sendMapRequest` is fire-and-forget

As noted in Task 8 Step 9, `sendMapRequest` only sends; the response arrives asynchronously via `forwardServerEvents`. The `handleMapModeInput` function sends the request and returns; `forwardServerEvents` handles the incoming `MapResponse` by updating `lastMapResponse` and calling `redrawMap`.

### Field number for `TravelRequest` in `ClientMessage`

Task 3 uses field number **106** because field 87 is occupied by `browse`. This is an implementation delta from the spec (which was written before `browse` was added at 87). The server-side `ClientMessage_Travel` case will use the generated `*gamev1.ClientMessage_Travel` type regardless of the field number.
