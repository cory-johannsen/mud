> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add recipe-driven crafting with material inventory, scavenging, NPC loot drops, merchant material stock, and quick/downtime craft execution.

**Architecture:** Three new pure packages — `internal/game/crafting/` (recipe + material registries, engine), used by `grpc_service.go` handlers. Material inventory persists in a new `character_materials` DB table loaded into `PlayerSession.Materials` at login. NPC material drops extend `FloorManager`. Downtime handoff via `DowntimeCraftStarter` interface (implemented by the `downtime` plan).

**Tech Stack:** Go, PostgreSQL, protobuf (buf), testify, rapid

**Dependencies:** `wire-refactor` (ContentDeps/StorageDeps pattern already implemented), `non-combat-npcs` (merchant material stock), `downtime` (craft handoff interface — stubbed here)

```go
// internal/game/crafting/materials.go
package crafting

import (
    "fmt"
    "os"
    "gopkg.in/yaml.v3"
)

type Material struct {
    ID       string `yaml:"id"`
    Name     string `yaml:"name"`
    Category string `yaml:"category"`
    Value    int    `yaml:"value"`
}

type materialFile struct {
    Materials []Material `yaml:"materials"`
}

type MaterialRegistry struct {
    materials map[string]*Material
}

func LoadMaterialRegistry(path string) (*MaterialRegistry, error) {
    data, err := os.ReadFile(path)
    if err != nil {
        return nil, fmt.Errorf("read materials: %w", err)
    }
    var f materialFile
    if err := yaml.Unmarshal(data, &f); err != nil {
        return nil, fmt.Errorf("parse materials: %w", err)
    }
    reg := &MaterialRegistry{materials: make(map[string]*Material, len(f.Materials))}
    for i := range f.Materials {
        m := &f.Materials[i]
        reg.materials[m.ID] = m
    }
    return reg, nil
}

func (r *MaterialRegistry) Material(id string) (*Material, bool) {
    m, ok := r.materials[id]
    return m, ok
}

func (r *MaterialRegistry) All() []*Material {
    out := make([]*Material, 0, len(r.materials))
    for _, m := range r.materials {
        out = append(out, m)
    }
    return out
}

func (r *MaterialRegistry) HasID(id string) bool {
    _, ok := r.materials[id]
    return ok
}
```

```go
// internal/game/crafting/recipe.go
package crafting

import (
    "fmt"
    "os"
    "path/filepath"
    "gopkg.in/yaml.v3"
)

type RecipeMaterial struct {
    ID       string `yaml:"id"`
    Quantity int    `yaml:"quantity"`
}

type Recipe struct {
    ID                string           `yaml:"id"`
    Name              string           `yaml:"name"`
    OutputItemID      string           `yaml:"output_item_id"`
    OutputCount       int              `yaml:"output_count"`
    Category          string           `yaml:"category"` // consumable|weapon|armor|item
    Complexity        int              `yaml:"complexity"`
    DC                int              `yaml:"dc"`
    QuickCraftMinRank string           `yaml:"quick_craft_min_rank"` // "never" = downtime only
    Materials         []RecipeMaterial `yaml:"materials"`
    Description       string           `yaml:"description"`
}

// EffectiveMinRank returns the quick-craft minimum rank, applying complexity defaults if unset.
func (r *Recipe) EffectiveMinRank() string {
    if r.QuickCraftMinRank != "" {
        return r.QuickCraftMinRank
    }
    switch r.Complexity {
    case 1:
        return "untrained"
    case 2:
        return "trained"
    case 3:
        return "expert"
    default:
        return "master"
    }
}

// DowntimeDays returns the days required for downtime crafting by complexity.
func (r *Recipe) DowntimeDays() int {
    switch r.Complexity {
    case 2:
        return 1
    case 3:
        return 2
    default: // 4
        return 4
    }
}

type RecipeRegistry struct {
    recipes map[string]*Recipe
}

// LoadRecipeRegistry loads all YAML files from dir and validates material + output IDs.
// matReg must be non-nil. invReg may be nil (skips output_item_id validation).
func LoadRecipeRegistry(dir string, matReg *MaterialRegistry, invReg OutputValidator) (*RecipeRegistry, error) {
    entries, err := os.ReadDir(dir)
    if err != nil {
        return nil, fmt.Errorf("read recipes dir: %w", err)
    }
    reg := &RecipeRegistry{recipes: make(map[string]*Recipe)}
    for _, e := range entries {
        if e.IsDir() || filepath.Ext(e.Name()) != ".yaml" {
            continue
        }
        data, err := os.ReadFile(filepath.Join(dir, e.Name()))
        if err != nil {
            return nil, err
        }
        var r Recipe
        if err := yaml.Unmarshal(data, &r); err != nil {
            return nil, fmt.Errorf("parse recipe %s: %w", e.Name(), err)
        }
        // REQ-CRAFT-6: validate material IDs
        for _, m := range r.Materials {
            if !matReg.HasID(m.ID) {
                return nil, fmt.Errorf("recipe %s: unknown material ID %q (REQ-CRAFT-6)", r.ID, m.ID)
            }
        }
        // REQ-CRAFT-10: validate output_item_id
        if invReg != nil {
            if err := invReg.ValidateOutput(r.Category, r.OutputItemID); err != nil {
                return nil, fmt.Errorf("recipe %s: %w (REQ-CRAFT-10)", r.ID, err)
            }
        }
        reg.recipes[r.ID] = &r
    }
    return reg, nil
}

// OutputValidator validates output_item_id against the inventory registry.
// Implemented by inventory.Registry wrapper.
type OutputValidator interface {
    ValidateOutput(category, itemID string) error
}

func (r *RecipeRegistry) Recipe(id string) (*Recipe, bool) {
    rec, ok := r.recipes[id]
    return rec, ok
}

func (r *RecipeRegistry) All() []*Recipe {
    out := make([]*Recipe, 0, len(r.recipes))
    for _, rec := range r.recipes {
        out = append(out, rec)
    }
    return out
}
```

```go
// internal/game/crafting/engine.go
package crafting

import "context"

// Outcome mirrors skillcheck.CheckOutcome for crafting results.
type Outcome int

const (
    CritSuccess Outcome = iota
    Success
    Failure
    CritFailure
)

// CraftResult is the pure output of a quick craft attempt.
type CraftResult struct {
    OutputQuantity    int
    MaterialsDeducted map[string]int // materialID → quantity to deduct
}

// DowntimeCraftStarter is implemented by the downtime feature.
type DowntimeCraftStarter interface {
    BeginCraftActivity(ctx context.Context, characterID int64, recipeID string, daysRequired int) error
}

type CraftingEngine struct{}

func NewEngine() *CraftingEngine { return &CraftingEngine{} }

// ExecuteQuickCraft computes the outcome of an immediate craft attempt.
// Precondition: recipe and playerMaterials are non-nil.
// Postcondition: returns deduction map and output count; does NOT mutate any state.
func (e *CraftingEngine) ExecuteQuickCraft(recipe *Recipe, playerMaterials map[string]int, outcome Outcome) CraftResult {
    switch outcome {
    case CritSuccess:
        return CraftResult{
            OutputQuantity:    recipe.OutputCount + 1,
            MaterialsDeducted: fullDeduction(recipe),
        }
    case Success:
        return CraftResult{
            OutputQuantity:    recipe.OutputCount,
            MaterialsDeducted: fullDeduction(recipe),
        }
    case Failure:
        return CraftResult{
            OutputQuantity:    0,
            MaterialsDeducted: halfDeduction(recipe),
        }
    default: // CritFailure
        return CraftResult{
            OutputQuantity:    0,
            MaterialsDeducted: fullDeduction(recipe),
        }
    }
}

func fullDeduction(r *Recipe) map[string]int {
    d := make(map[string]int, len(r.Materials))
    for _, m := range r.Materials {
        d[m.ID] = m.Quantity
    }
    return d
}

func halfDeduction(r *Recipe) map[string]int {
    d := make(map[string]int, len(r.Materials))
    for _, m := range r.Materials {
        d[m.ID] = m.Quantity / 2 // floor per material (REQ-CRAFT-5)
    }
    return d
}
```

```go
// internal/storage/postgres/character_materials.go
package postgres

import (
    "context"
    "database/sql"
    "fmt"
)

type CharacterMaterialsRepository struct {
    db *sql.DB
}

func NewCharacterMaterialsRepository(db *sql.DB) *CharacterMaterialsRepository {
    return &CharacterMaterialsRepository{db: db}
}

// Load returns all materials for a character as a materialID→quantity map.
func (r *CharacterMaterialsRepository) Load(ctx context.Context, characterID int64) (map[string]int, error) {
    rows, err := r.db.QueryContext(ctx,
        `SELECT material_id, quantity FROM character_materials WHERE character_id = $1`,
        characterID)
    if err != nil {
        return nil, err
    }
    defer rows.Close()
    result := make(map[string]int)
    for rows.Next() {
        var id string
        var qty int
        if err := rows.Scan(&id, &qty); err != nil {
            return nil, err
        }
        result[id] = qty
    }
    return result, rows.Err()
}

// Add increments a material quantity, inserting a new row if needed.
func (r *CharacterMaterialsRepository) Add(ctx context.Context, characterID int64, materialID string, amount int) error {
    _, err := r.db.ExecContext(ctx, `
        INSERT INTO character_materials (character_id, material_id, quantity)
        VALUES ($1, $2, $3)
        ON CONFLICT (character_id, material_id)
        DO UPDATE SET quantity = character_materials.quantity + EXCLUDED.quantity`,
        characterID, materialID, amount)
    return err
}

// DeductMany executes all deductions in a single transaction. (REQ-CRAFT-12)
// Rows where quantity reaches zero are deleted.
func (r *CharacterMaterialsRepository) DeductMany(ctx context.Context, characterID int64, deductions map[string]int) error {
    tx, err := r.db.BeginTx(ctx, nil)
    if err != nil {
        return err
    }
    defer tx.Rollback()
    for materialID, qty := range deductions {
        if qty == 0 {
            continue
        }
        // Decrement; if new qty <= 0 delete row; CHECK constraint also enforces qty > 0
        res, err := tx.ExecContext(ctx, `
            UPDATE character_materials
            SET quantity = quantity - $1
            WHERE character_id = $2 AND material_id = $3 AND quantity >= $1`,
            qty, characterID, materialID)
        if err != nil {
            return err
        }
        n, _ := res.RowsAffected()
        if n == 0 {
            return fmt.Errorf("insufficient %s", materialID)
        }
        // Delete rows where quantity is now 0
        _, err = tx.ExecContext(ctx,
            `DELETE FROM character_materials WHERE character_id = $1 AND material_id = $2 AND quantity = 0`,
            characterID, materialID)
        if err != nil {
            return err
        }
    }
    return tx.Commit()
}
```

```go
func bridgeCraft(bctx *bridgeContext) (bridgeResult, error) {
    args := strings.Fields(bctx.parsed.RawArgs)
    if len(args) == 0 {
        // Treat bare "craft" as "craft list"
        return bridgeResult{msg: &gamev1.ClientMessage{
            RequestId: bctx.reqID,
            Payload:   &gamev1.ClientMessage_CraftList{CraftList: &gamev1.CraftListRequest{}},
        }}, nil
    }
    switch strings.ToLower(args[0]) {
```

```go
func (s *GameServiceServer) handleScavenge(uid string) error {
    sess, ok := s.sessions.Get(uid)
    if !ok {
        return fmt.Errorf("session not found")
    }
    // REQ-CRAFT-11: one attempt per room visit
    if sess.ScavengeExhaustedRoomID == sess.RoomID {
        return s.sendError(uid, "You've already scavenged this area.")
    }
    sess.ScavengeExhaustedRoomID = sess.RoomID

    // Room.ZoneID holds the zone ID directly (world.Room.ZoneID field).
    room, roomOK := s.world.GetRoom(sess.RoomID)
    var zone *world.Zone
    var zoneOK bool
    if roomOK {
        zone, zoneOK = s.world.GetZone(room.ZoneID)
    }
    if !roomOK || !zoneOK || zone.MaterialPool == nil {
        return s.sendConsole(uid, "There's nothing useful to scavenge here.") // REQ-CRAFT-9
    }

    // Scavenging skill check
    pool := zone.MaterialPool
    scavRank := sess.Skills["scavenging"]
    if scavRank == "" {
        scavRank = "untrained"
    }
    roll := s.dice.Roll(1, 20, 0)
    abilityMod := (sess.Abilities.Savvy - 10) / 2 // Scavenging uses Savvy
    result := skillcheck.Resolve(roll, abilityMod, scavRank, pool.DC, skillcheck.TriggerDef{})

    count := 0
    switch result.Outcome {
    case skillcheck.CritSuccess:
        count = 3
    case skillcheck.Success:
        count = 1 + s.dice.Roll(1, 2, 0) - 1 // 1-2
    default:
        return s.sendConsole(uid, "You scavenge the area but find nothing useful.")
    }

    gained := s.drawFromPool(pool.Drops, count)
    for matID, qty := range gained {
        sess.Materials[matID] += qty
        if s.materialRepo != nil {
            _ = s.materialRepo.Add(context.Background(), sess.CharacterID, matID, qty)
        }
    }
    // Format and send console message listing gained materials
    return s.sendConsole(uid, s.formatScavengeResult(gained))
}
```
