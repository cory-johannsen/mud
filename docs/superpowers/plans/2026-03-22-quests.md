// internal/game/quest/def.go
package quest

import (
	"fmt"
	"time"
)

// ValidObjectiveTypes is the set of allowed objective type strings.
var ValidObjectiveTypes = map[string]bool{
	"kill":    true,
	"fetch":   true,
	"explore": true,
	"deliver": true,
}

// QuestRewardItem specifies an item and quantity awarded on quest completion.
type QuestRewardItem struct {
	ItemID   string `yaml:"item_id"`
	Quantity int    `yaml:"quantity"`
}

// QuestRewards holds XP, credit, and item rewards granted on completion.
type QuestRewards struct {
	XP      int               `yaml:"xp"`
	Credits int               `yaml:"credits"`
	Items   []QuestRewardItem `yaml:"items"`
}

// QuestObjective is one step within a quest definition.
//
// TargetID semantics by Type:
//   - kill:    NPC template ID
//   - fetch:   item definition ID
//   - explore: room ID
//   - deliver: NPC template ID (ItemID is the item to bring)
type QuestObjective struct {
	ID          string `yaml:"id"`
	Type        string `yaml:"type"`
	Description string `yaml:"description"`
	TargetID    string `yaml:"target_id"`
	ItemID      string `yaml:"item_id,omitempty"`
	Quantity    int    `yaml:"quantity"`
}

// QuestDef is the static definition of a quest loaded from YAML.
type QuestDef struct {
	ID            string          `yaml:"id"`
	Title         string          `yaml:"title"`
	Description   string          `yaml:"description"`
	GiverNPCID    string          `yaml:"giver_npc_id"`
	Repeatable    bool            `yaml:"repeatable"`
	Cooldown      string          `yaml:"cooldown,omitempty"`
	Prerequisites []string        `yaml:"prerequisites,omitempty"`
	Objectives    []QuestObjective `yaml:"objectives"`
	Rewards       QuestRewards    `yaml:"rewards"`
}

// Validate checks all required fields and invariants.
//
// Precondition: none.
// Postcondition: Returns nil if and only if the QuestDef is structurally valid.
func (d *QuestDef) Validate() error {
	if d.ID == "" {
		return fmt.Errorf("quest ID must not be empty")
	}
	if d.Title == "" {
		return fmt.Errorf("quest %q: Title must not be empty", d.ID)
	}
	if d.GiverNPCID == "" {
		return fmt.Errorf("quest %q: GiverNPCID must not be empty", d.ID)
	}
	if len(d.Objectives) == 0 {
		return fmt.Errorf("quest %q: Objectives must not be empty", d.ID)
	}
	for _, obj := range d.Objectives {
		if obj.ID == "" {
			return fmt.Errorf("quest %q: objective ID must not be empty", d.ID)
		}
		if obj.Description == "" {
			return fmt.Errorf("quest %q objective %q: Description must not be empty", d.ID, obj.ID)
		}
		if obj.TargetID == "" {
			return fmt.Errorf("quest %q objective %q: TargetID must not be empty", d.ID, obj.ID)
		}
		if !ValidObjectiveTypes[obj.Type] {
			return fmt.Errorf("quest %q objective %q: invalid Type %q", d.ID, obj.ID, obj.Type)
		}
		if obj.Quantity < 1 {
			return fmt.Errorf("quest %q objective %q: Quantity must be >= 1", d.ID, obj.ID)
		}
		if obj.Type == "deliver" && obj.ItemID == "" {
			return fmt.Errorf("quest %q objective %q: deliver objective requires ItemID", d.ID, obj.ID)
		}
	}
	if !d.Repeatable && d.Cooldown != "" {
		return fmt.Errorf("quest %q: non-repeatable quest must not have Cooldown", d.ID)
	}
	if d.Cooldown != "" {
		if _, err := time.ParseDuration(d.Cooldown); err != nil {
			return fmt.Errorf("quest %q: invalid Cooldown %q: %w", d.ID, d.Cooldown, err)
		}
	}
	return nil
}
```

// internal/game/quest/registry.go
package quest

import (
	"fmt"
	"os"
	"path/filepath"

	"gopkg.in/yaml.v3"
)

// QuestRegistry is a map from quest ID to its definition.
type QuestRegistry map[string]*QuestDef

// LoadFromDir reads all *.yaml files in dir, parses each as a QuestDef,
// validates it, and returns the populated registry.
//
// Precondition: dir must be a readable directory path.
// Postcondition: All returned entries have passed Validate(); error is non-nil on any failure.
func LoadFromDir(dir string) (QuestRegistry, error) {
	entries, err := os.ReadDir(dir)
	if err != nil {
		return nil, fmt.Errorf("reading quest dir %q: %w", dir, err)
	}
	reg := make(QuestRegistry)
	for _, e := range entries {
		if e.IsDir() || filepath.Ext(e.Name()) != ".yaml" {
			continue
		}
		path := filepath.Join(dir, e.Name())
		data, err := os.ReadFile(path)
		if err != nil {
			return nil, fmt.Errorf("reading quest file %q: %w", path, err)
		}
		var def QuestDef
		if err := yaml.Unmarshal(data, &def); err != nil {
			return nil, fmt.Errorf("parsing quest file %q: %w", path, err)
		}
		if err := def.Validate(); err != nil {
			return nil, fmt.Errorf("invalid quest in %q: %w", path, err)
		}
		reg[def.ID] = &def
	}
	return reg, nil
}

// CrossValidate checks all quest definition references against live registries.
// npcIDs is the set of all known NPC template IDs; itemIDs the set of item def IDs;
// roomIDs the set of all room IDs in the world.
//
// Precondition: all maps must be non-nil (may be empty).
// Postcondition: Returns a fatal error if any reference is unresolvable.
func (r QuestRegistry) CrossValidate(npcIDs, itemIDs, roomIDs map[string]bool) error {
	for _, def := range r {
		if !npcIDs[def.GiverNPCID] {
			return fmt.Errorf("quest %q: GiverNPCID %q not found in NPC registry", def.ID, def.GiverNPCID)
		}
		for _, prereq := range def.Prerequisites {
			if _, ok := r[prereq]; !ok {
				return fmt.Errorf("quest %q: prerequisite quest %q not found in QuestRegistry", def.ID, prereq)
			}
		}
		for _, obj := range def.Objectives {
			switch obj.Type {
			case "kill":
				if !npcIDs[obj.TargetID] {
					return fmt.Errorf("quest %q objective %q: kill TargetID %q not in NPC registry", def.ID, obj.ID, obj.TargetID)
				}
			case "fetch":
				if !itemIDs[obj.TargetID] {
					return fmt.Errorf("quest %q objective %q: fetch TargetID %q not in item registry", def.ID, obj.ID, obj.TargetID)
				}
			case "explore":
				if !roomIDs[obj.TargetID] {
					return fmt.Errorf("quest %q objective %q: explore TargetID %q not in world rooms", def.ID, obj.ID, obj.TargetID)
				}
			case "deliver":
				if !npcIDs[obj.TargetID] {
					return fmt.Errorf("quest %q objective %q: deliver TargetID %q not in NPC registry", def.ID, obj.ID, obj.TargetID)
				}
				if !itemIDs[obj.ItemID] {
					return fmt.Errorf("quest %q objective %q: deliver ItemID %q not in item registry", def.ID, obj.ID, obj.ItemID)
				}
			}
		}
	}
	return nil
}
```

// internal/game/quest/types.go
package quest

import (
	"context"
	"time"
)

// ActiveQuest is the in-session state of a quest the player is currently on.
type ActiveQuest struct {
	QuestID           string
	ObjectiveProgress map[string]int // objective ID → count completed so far
}

// QuestRecord is the persisted row returned from LoadQuests.
type QuestRecord struct {
	CharacterID int64
	QuestID     string
	Status      string // "active" | "completed" | "abandoned"
	Progress    map[string]int
	CompletedAt *time.Time
}

// QuestRepository is the storage interface for quest persistence.
//
// Precondition: characterID > 0; questID and objectiveID must be non-empty.
type QuestRepository interface {
	// SaveQuestStatus upserts the character_quests row.
	SaveQuestStatus(ctx context.Context, characterID int64, questID, status string, completedAt *time.Time) error
	// SaveObjectiveProgress upserts the character_quest_progress row.
	SaveObjectiveProgress(ctx context.Context, characterID int64, questID, objectiveID string, progress int) error
	// LoadQuests returns all quest rows for a character (join of both tables).
	LoadQuests(ctx context.Context, characterID int64) ([]QuestRecord, error)
}

// QuestRepository implements quest.QuestRepository backed by PostgreSQL.
type QuestRepository struct {
	db *pgxpool.Pool
}

// NewQuestRepository creates a new QuestRepository.
//
// Precondition: db must be non-nil.
func NewQuestRepository(db *pgxpool.Pool) *QuestRepository {
	return &QuestRepository{db: db}
}

// SaveQuestStatus upserts the character_quests row for (characterID, questID).
func (r *QuestRepository) SaveQuestStatus(ctx context.Context, characterID int64, questID, status string, completedAt *time.Time) error {
	_, err := r.db.Exec(ctx, `
		INSERT INTO character_quests (character_id, quest_id, status, completed_at)
		VALUES ($1, $2, $3, $4)
		ON CONFLICT (character_id, quest_id) DO UPDATE
		  SET status = EXCLUDED.status, completed_at = EXCLUDED.completed_at
	`, characterID, questID, status, completedAt)
	if err != nil {
		return fmt.Errorf("SaveQuestStatus: %w", err)
	}
	return nil
}

// SaveObjectiveProgress upserts the character_quest_progress row.
func (r *QuestRepository) SaveObjectiveProgress(ctx context.Context, characterID int64, questID, objectiveID string, progress int) error {
	_, err := r.db.Exec(ctx, `
		INSERT INTO character_quest_progress (character_id, quest_id, objective_id, progress)
		VALUES ($1, $2, $3, $4)
		ON CONFLICT (character_id, quest_id, objective_id) DO UPDATE
		  SET progress = EXCLUDED.progress
	`, characterID, questID, objectiveID, progress)
	if err != nil {
		return fmt.Errorf("SaveObjectiveProgress: %w", err)
	}
	return nil
}

// LoadQuests returns all quest records for a character by joining both tables.
// Objectives with no progress rows produce an empty Progress map.
func (r *QuestRepository) LoadQuests(ctx context.Context, characterID int64) ([]quest.QuestRecord, error) {
	rows, err := r.db.Query(ctx, `
		SELECT cq.quest_id, cq.status, cq.completed_at,
		       coalesce(cqp.objective_id, '') AS objective_id,
		       coalesce(cqp.progress, 0)      AS progress
		FROM character_quests cq
		LEFT JOIN character_quest_progress cqp
		  ON cq.character_id = cqp.character_id AND cq.quest_id = cqp.quest_id
		WHERE cq.character_id = $1
		ORDER BY cq.quest_id, cqp.objective_id
	`, characterID)
	if err != nil {
		return nil, fmt.Errorf("LoadQuests query: %w", err)
	}
	defer rows.Close()

	byID := make(map[string]*quest.QuestRecord)
	var order []string
	for rows.Next() {
		var questID, status, objID string
		var completedAt *time.Time
		var prog int
		if err := rows.Scan(&questID, &status, &completedAt, &objID, &prog); err != nil {
			return nil, fmt.Errorf("LoadQuests scan: %w", err)
		}
		rec, exists := byID[questID]
		if !exists {
			rec = &quest.QuestRecord{
				CharacterID: characterID,
				QuestID:     questID,
				Status:      status,
				CompletedAt: completedAt,
				Progress:    make(map[string]int),
			}
			byID[questID] = rec
			order = append(order, questID)
		}
		if objID != "" {
			rec.Progress[objID] = prog
		}
	}
	if err := rows.Err(); err != nil {
		return nil, fmt.Errorf("LoadQuests rows: %w", err)
	}
	result := make([]quest.QuestRecord, 0, len(order))
	for _, id := range order {
		result = append(result, *byID[id])
	}
	return result, nil
}

package quest

import (
	"context"
	"fmt"
	"time"

	"github.com/cory-johannsen/mud/internal/game/inventory"
	"github.com/cory-johannsen/mud/internal/game/session"
	"github.com/cory-johannsen/mud/internal/game/xp"
)

// Service is the quest business logic layer.
//
// Precondition: registry and repo must be non-nil; xpSvc, invRegistry, and charSaver may be nil
// (quest completion requiring those will no-op gracefully for unit tests).
type Service struct {
	registry    QuestRegistry
	repo        QuestRepository
	xpSvc       XPAwarder
	invRegistry *inventory.Registry
	charSaver   InventorySaver
}

// XPAwarder abstracts xp.Service.AwardXPAmount for testability.
type XPAwarder interface {
	AwardXPAmount(ctx context.Context, sess *session.PlayerSession, characterID int64, amount int) ([]string, error)
}

// InventorySaver abstracts the character inventory persistence call.
type InventorySaver interface {
	SaveInventory(ctx context.Context, characterID int64, items []inventory.InventoryItem) error
	SaveCurrency(ctx context.Context, characterID int64, currency int) error
}

// NewService creates a new quest Service.
func NewService(registry QuestRegistry, repo QuestRepository, xpSvc XPAwarder, invRegistry *inventory.Registry, charSaver InventorySaver) *Service {
	return &Service{
		registry:    registry,
		repo:        repo,
		xpSvc:       xpSvc,
		invRegistry: invRegistry,
		charSaver:   charSaver,
	}
}

// HydrateSession populates sess.ActiveQuests and sess.CompletedQuests from DB records.
//
// Precondition: records are from LoadQuests; sess is non-nil with maps initialized.
// Postcondition: session maps reflect persisted state per REQ-QU-14.
func (s *Service) HydrateSession(sess *session.PlayerSession, records []QuestRecord) {
	for _, r := range records {
		def, ok := s.registry[r.QuestID]
		switch r.Status {
		case "active":
			sess.ActiveQuests[r.QuestID] = &ActiveQuest{
				QuestID:           r.QuestID,
				ObjectiveProgress: r.Progress,
			}
		case "completed":
			t := r.CompletedAt
			sess.CompletedQuests[r.QuestID] = t
		case "abandoned":
			if ok && !def.Repeatable {
				// nil sentinel blocks re-acceptance
				sess.CompletedQuests[r.QuestID] = nil
			}
			// repeatable abandoned quests are not loaded (can be re-offered)
		}
		_ = ok
	}
}

// GetOfferable returns quests from questIDs that are eligible to offer to sess.
//
// Eligibility rules (REQ-QU-16):
//  1. Not currently in ActiveQuests.
//  2. All prerequisites in CompletedQuests with non-nil completedAt.
//  3. If non-repeatable: not in CompletedQuests at all.
//  4. If repeatable: not in CompletedQuests or cooldown elapsed.
func (s *Service) GetOfferable(sess *session.PlayerSession, questIDs []string) []*QuestDef {
	var result []*QuestDef
	for _, qid := range questIDs {
		def, ok := s.registry[qid]
		if !ok {
			continue
		}
		if _, active := sess.ActiveQuests[qid]; active {
			continue
		}
		// Prerequisites check
		eligible := true
		for _, prereq := range def.Prerequisites {
			completedAt, present := sess.CompletedQuests[prereq]
			if !present || completedAt == nil {
				eligible = false
				break
			}
		}
		if !eligible {
			continue
		}
		if !def.Repeatable {
			if _, done := sess.CompletedQuests[qid]; done {
				continue
			}
		} else {
			if completedAt, done := sess.CompletedQuests[qid]; done && completedAt != nil {
				if def.Cooldown != "" {
					cd, _ := time.ParseDuration(def.Cooldown)
					if time.Since(*completedAt) < cd {
						continue
					}
				}
			}
		}
		result = append(result, def)
	}
	return result
}

// Accept adds a quest to the player's active quests after re-checking eligibility.
//
// Precondition: questID must exist in registry; sess and ctx must be non-nil.
// Postcondition: DB rows inserted; sess.ActiveQuests updated; returns title and objective descriptions.
func (s *Service) Accept(ctx context.Context, sess *session.PlayerSession, characterID int64, questID string) (title string, objDescs []string, err error) {
	def, ok := s.registry[questID]
	if !ok {
		return "", nil, fmt.Errorf("quest %q not found", questID)
	}
	offerable := s.GetOfferable(sess, []string{questID})
	if len(offerable) == 0 {
		return "", nil, fmt.Errorf("quest %q is not available to you", questID)
	}
	// Insert character_quests row
	if err := s.repo.SaveQuestStatus(ctx, characterID, questID, "active", nil); err != nil {
		return "", nil, fmt.Errorf("saving quest status: %w", err)
	}
	// Insert per-objective progress rows
	for _, obj := range def.Objectives {
		if err := s.repo.SaveObjectiveProgress(ctx, characterID, questID, obj.ID, 0); err != nil {
			return "", nil, fmt.Errorf("saving objective progress: %w", err)
		}
	}
	// Update session
	progress := make(map[string]int, len(def.Objectives))
	for _, obj := range def.Objectives {
		progress[obj.ID] = 0
	}
	sess.ActiveQuests[questID] = &ActiveQuest{QuestID: questID, ObjectiveProgress: progress}
	descs := make([]string, len(def.Objectives))
	for i, obj := range def.Objectives {
		descs[i] = obj.Description
	}
	return def.Title, descs, nil
}

// RecordKill increments kill objective progress for all active quests targeting npcTemplateID.
//
// Precondition: sess non-nil; npcTemplateID non-empty.
// Postcondition: Progress persisted; maybeComplete called for each updated quest.
func (s *Service) RecordKill(ctx context.Context, sess *session.PlayerSession, characterID int64, npcTemplateID string) error {
	for qid, aq := range sess.ActiveQuests {
		def, ok := s.registry[qid]
		if !ok {
			continue
		}
		updated := false
		for _, obj := range def.Objectives {
			if obj.Type != "kill" || obj.TargetID != npcTemplateID {
				continue
			}
			if aq.ObjectiveProgress[obj.ID] < obj.Quantity {
				aq.ObjectiveProgress[obj.ID]++
				if err := s.repo.SaveObjectiveProgress(ctx, characterID, qid, obj.ID, aq.ObjectiveProgress[obj.ID]); err != nil {
					return fmt.Errorf("persisting kill progress: %w", err)
				}
				updated = true
			}
		}
		if updated {
			if err := s.maybeComplete(ctx, sess, characterID, qid); err != nil {
				return err
			}
		}
	}
	return nil
}

// RecordFetch increments fetch objective progress for active quests matching itemDefID.
func (s *Service) RecordFetch(ctx context.Context, sess *session.PlayerSession, characterID int64, itemDefID string, qty int) error {
	for qid, aq := range sess.ActiveQuests {
		def, ok := s.registry[qid]
		if !ok {
			continue
		}
		updated := false
		for _, obj := range def.Objectives {
			if obj.Type != "fetch" || obj.TargetID != itemDefID {
				continue
			}
			newProg := aq.ObjectiveProgress[obj.ID] + qty
			if newProg > obj.Quantity {
				newProg = obj.Quantity
			}
			if newProg != aq.ObjectiveProgress[obj.ID] {
				aq.ObjectiveProgress[obj.ID] = newProg
				if err := s.repo.SaveObjectiveProgress(ctx, characterID, qid, obj.ID, newProg); err != nil {
					return fmt.Errorf("persisting fetch progress: %w", err)
				}
				updated = true
			}
		}
		if updated {
			if err := s.maybeComplete(ctx, sess, characterID, qid); err != nil {
				return err
			}
		}
	}
	return nil
}

// RecordExplore sets explore objective progress to Quantity (1) for active quests matching roomID.
func (s *Service) RecordExplore(ctx context.Context, sess *session.PlayerSession, characterID int64, roomID string) error {
	for qid, aq := range sess.ActiveQuests {
		def, ok := s.registry[qid]
		if !ok {
			continue
		}
		updated := false
		for _, obj := range def.Objectives {
			if obj.Type != "explore" || obj.TargetID != roomID {
				continue
			}
			if aq.ObjectiveProgress[obj.ID] < obj.Quantity {
				aq.ObjectiveProgress[obj.ID] = obj.Quantity
				if err := s.repo.SaveObjectiveProgress(ctx, characterID, qid, obj.ID, obj.Quantity); err != nil {
					return fmt.Errorf("persisting explore progress: %w", err)
				}
				updated = true
			}
		}
		if updated {
			if err := s.maybeComplete(ctx, sess, characterID, qid); err != nil {
				return err
			}
		}
	}
	return nil
}

// RecordDeliver marks a deliver objective complete (progress = Quantity) for the given NPC.
// Returns true if the objective was updated.
func (s *Service) RecordDeliver(ctx context.Context, sess *session.PlayerSession, characterID int64, questID, objectiveID string) error {
	aq, ok := sess.ActiveQuests[questID]
	if !ok {
		return fmt.Errorf("quest %q not active", questID)
	}
	def, ok := s.registry[questID]
	if !ok {
		return fmt.Errorf("quest %q not in registry", questID)
	}
	for _, obj := range def.Objectives {
		if obj.ID != objectiveID {
			continue
		}
		aq.ObjectiveProgress[obj.ID] = obj.Quantity
		if err := s.repo.SaveObjectiveProgress(ctx, characterID, questID, obj.ID, obj.Quantity); err != nil {
			return fmt.Errorf("persisting deliver progress: %w", err)
		}
		return s.maybeComplete(ctx, sess, characterID, questID)
	}
	return fmt.Errorf("objective %q not found in quest %q", objectiveID, questID)
}

// maybeComplete checks if all objectives are satisfied and triggers Complete if so.
func (s *Service) maybeComplete(ctx context.Context, sess *session.PlayerSession, characterID int64, questID string) error {
	aq, ok := sess.ActiveQuests[questID]
	if !ok {
		return nil
	}
	def, ok := s.registry[questID]
	if !ok {
		return nil
	}
	for _, obj := range def.Objectives {
		if aq.ObjectiveProgress[obj.ID] < obj.Quantity {
			return nil
		}
	}
	return s.Complete(ctx, sess, characterID, questID)
}

// Complete finalizes a quest, awards rewards, and updates session/DB.
//
// Precondition: questID must be active in sess; all objectives must be satisfied.
// Postcondition: quest moved to CompletedQuests; rewards applied; DB updated.
func (s *Service) Complete(ctx context.Context, sess *session.PlayerSession, characterID int64, questID string) error {
	def, ok := s.registry[questID]
	if !ok {
		return fmt.Errorf("quest %q not in registry", questID)
	}
	now := time.Now().UTC()
	if err := s.repo.SaveQuestStatus(ctx, characterID, questID, "completed", &now); err != nil {
		return fmt.Errorf("completing quest in DB: %w", err)
	}
	// Award XP
	if s.xpSvc != nil && def.Rewards.XP > 0 {
		_, _ = s.xpSvc.AwardXPAmount(ctx, sess, characterID, def.Rewards.XP)
	}
	// Award items (collect overflow to send to caller — caller drops to floor)
	if s.invRegistry != nil && len(def.Rewards.Items) > 0 {
		for _, ri := range def.Rewards.Items {
			_, _ = sess.Backpack.Add(ri.ItemID, ri.Quantity, s.invRegistry)
		}
		if s.charSaver != nil {
			items := backpackToInventoryItems(sess.Backpack)
			_ = s.charSaver.SaveInventory(ctx, characterID, items)
		}
	}
	// Award credits
	if def.Rewards.Credits > 0 {
		sess.Currency += def.Rewards.Credits
		if s.charSaver != nil {
			_ = s.charSaver.SaveCurrency(ctx, characterID, sess.Currency)
		}
	}
	// Update session maps
	delete(sess.ActiveQuests, questID)
	sess.CompletedQuests[questID] = &now
	return nil
}

// Abandon removes a quest from active quests.
// If non-repeatable and confirm is false, returns a confirmation prompt.
// Returns (message, nil) — the caller should display the message.
func (s *Service) Abandon(ctx context.Context, sess *session.PlayerSession, characterID int64, questID string, confirm bool) (string, error) {
	def, ok := s.registry[questID]
	if !ok {
		return "No active quest with that ID.", nil
	}
	if _, active := sess.ActiveQuests[questID]; !active {
		return "You don't have that quest.", nil
	}
	if !def.Repeatable && !confirm {
		return fmt.Sprintf("This quest cannot be retaken. Type 'quest abandon %s confirm' to proceed.", questID), nil
	}
	if err := s.repo.SaveQuestStatus(ctx, characterID, questID, "abandoned", nil); err != nil {
		return "", fmt.Errorf("abandoning quest in DB: %w", err)
	}
	delete(sess.ActiveQuests, questID)
	if !def.Repeatable {
		sess.CompletedQuests[questID] = nil // sentinel
	}
	return fmt.Sprintf("Quest abandoned: %s", def.Title), nil
}

// CompletionMessage formats the reward summary message per REQ-QU-25.
func CompletionMessage(def *QuestDef, invRegistry *inventory.Registry) []string {
	lines := []string{fmt.Sprintf("Quest complete: %s", def.Title)}
	rewardLine := fmt.Sprintf("+%d XP", def.Rewards.XP)
	if def.Rewards.Credits > 0 {
		rewardLine += fmt.Sprintf(" | +%d credits", def.Rewards.Credits)
	}
	lines = append(lines, rewardLine)
	for _, ri := range def.Rewards.Items {
		name := ri.ItemID
		if invRegistry != nil {
			if d, ok := invRegistry.Item(ri.ItemID); ok {
				name = d.Name
			}
		}
		lines = append(lines, fmt.Sprintf("+%dx %s", ri.Quantity, name))
	}
	return lines
}

// backpackToInventoryItems converts a Backpack to the slice type expected by SaveInventory.
func backpackToInventoryItems(bp *inventory.Backpack) []inventory.InventoryItem {
	items := bp.Items()
	result := make([]inventory.InventoryItem, len(items))
	for i, it := range items {
		result[i] = inventory.InventoryItem{InstanceID: it.InstanceID, ItemDefID: it.ItemDefID, Quantity: it.Quantity}
	}
	return result
}

					_ = sess.Backpack.Remove(m.InstanceID, toRemove)
					remaining -= toRemove
				}

					deliverMsgs = append(deliverMsgs, fmt.Sprintf("Deliver error: %v", err))
				} else {
					deliverMsgs = append(deliverMsgs, fmt.Sprintf("You hand over the items."))
				}
			}
		}
	}

	// Show offerable quests
	if s.questSvc != nil && len(tmpl.QuestGiver.QuestIDs) > 0 {
		offerable := s.questSvc.GetOfferable(sess, tmpl.QuestGiver.QuestIDs)
		if len(offerable) > 0 {

			for _, line := range deliverMsgs {

				sb.WriteString("\n")
			}
			for _, def := range offerable {
				sb.WriteString(fmt.Sprintf("[Quest] %s\n%s\nRewards: %d XP", def.Title, def.Description, def.Rewards.XP))
				if def.Rewards.Credits > 0 {
					sb.WriteString(fmt.Sprintf(", %d credits", def.Rewards.Credits))
				}
				sb.WriteString("\n")
			}
			sb.WriteString(fmt.Sprintf("Type 'talk %s accept <quest_id>' to accept.", inst.Name()))
			return messageEvent(sb.String()), nil
		}
	}

	// No offerable quests — placeholder dialog or deliver messages
	if len(deliverMsgs) > 0 {
		return messageEvent(strings.Join(deliverMsgs, "\n")), nil
	}
	dialog := tmpl.QuestGiver.PlaceholderDialog
	line := dialog[rand.Intn(len(dialog))]
	return messageEvent(fmt.Sprintf("%s says: %q", inst.Name(), line)), nil
}

// questList formats active quests with objective completion ratios (REQ-QU-29).
func (s *GameServiceServer) questList(sess *sessionpkg.PlayerSession) *gamev1.ServerEvent {
	if len(sess.ActiveQuests) == 0 {
		return messageEvent("You have no active quests.")
	}
	var sb strings.Builder
	for qid, aq := range sess.ActiveQuests {
		def, ok := s.questSvc.Registry()[qid]
		if !ok {
			continue
		}
		done := 0
		for _, obj := range def.Objectives {
			if aq.ObjectiveProgress[obj.ID] >= obj.Quantity {
				done++
			}
		}
		sb.WriteString(fmt.Sprintf("[%d/%d] %s — %s\n", done, len(def.Objectives), qid, def.Title))
	}
	return messageEvent(strings.TrimRight(sb.String(), "\n"))
}

// questLog shows full detail for an active or completed quest (REQ-QU-30).
func (s *GameServiceServer) questLog(sess *sessionpkg.PlayerSession, questID string) *gamev1.ServerEvent {
	reg := s.questSvc.Registry()
	def, ok := reg[questID]
	if !ok {
		return messageEvent("No quest found with that ID.")
	}
	var sb strings.Builder
	// Completed quest
	if completedAt, done := sess.CompletedQuests[questID]; done && completedAt != nil {
		sb.WriteString(fmt.Sprintf("(Completed: %s)\n", completedAt.Format("2006-01-02 15:04:05")))
		sb.WriteString(fmt.Sprintf("%s\n%s\n", def.Title, def.Description))
		for _, obj := range def.Objectives {
			sb.WriteString(fmt.Sprintf("[x] %s (%d/%d)\n", obj.Description, obj.Quantity, obj.Quantity))
		}
		return messageEvent(strings.TrimRight(sb.String(), "\n"))
	}
	// Active quest
	if aq, active := sess.ActiveQuests[questID]; active {
		sb.WriteString(fmt.Sprintf("%s\n%s\n", def.Title, def.Description))
		for _, obj := range def.Objectives {
			prog := aq.ObjectiveProgress[obj.ID]
			check := "[ ]"
			if prog >= obj.Quantity {
				check = "[x]"
			}
			sb.WriteString(fmt.Sprintf("%s %s (%d/%d)\n", check, obj.Description, prog, obj.Quantity))
		}
		// Rewards
		sb.WriteString(fmt.Sprintf("Rewards: %d XP", def.Rewards.XP))
		if def.Rewards.Credits > 0 {
			sb.WriteString(fmt.Sprintf(", %d credits", def.Rewards.Credits))
		}
		for _, ri := range def.Rewards.Items {
			name := ri.ItemID
			if s.invRegistry != nil {
				if d, ok := s.invRegistry.Item(ri.ItemID); ok {
					name = d.Name
				}
			}
			sb.WriteString(fmt.Sprintf(", %dx %s", ri.Quantity, name))
		}
		return messageEvent(strings.TrimRight(sb.String(), "\n"))
	}
	return messageEvent("No quest found with that ID.")
}

// questCompleted lists completed quests with timestamps (REQ-QU-31).
func (s *GameServiceServer) questCompleted(sess *sessionpkg.PlayerSession) *gamev1.ServerEvent {
	reg := s.questSvc.Registry()
	var sb strings.Builder
	found := false
	for qid, completedAt := range sess.CompletedQuests {
		if completedAt == nil {

		}

		}

		found = true
	}
	if !found {
		return messageEvent("You have not completed any quests.")
	}
	return messageEvent(strings.TrimRight(sb.String(), "\n"))
