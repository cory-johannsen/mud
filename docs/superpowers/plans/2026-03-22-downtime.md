# Downtime Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add 15 downtime activities that run while a player is in a Safe room, persist across disconnects, and resolve via skill checks after a timed delay.

**Architecture:** `internal/game/downtime/` holds pure activity definitions and engine logic. `DowntimeBusy bool` on `PlayerSession` blocks movement/combat via per-handler checks. The game clock tick calls `CheckCompletion()` every real minute. Activity state persists in `character_downtime` DB table and is restored on reconnect. All skill checks use the standard `skillcheck.Resolve` engine.

**Tech Stack:** Go, PostgreSQL, testify, rapid

**Dependencies:** `persistent-calendar` (GameClock tick), `crafting` (Craft activity uses CraftingEngine), `focus-points` (Recalibrate uses focuspoints.Restore)

```go
// internal/storage/postgres/character_downtime.go
package postgres

import (
    "context"
    "database/sql"
    "time"
)

// DowntimeState is the persisted form of an active downtime activity.
type DowntimeState struct {
    ActivityID  string
    CompletesAt time.Time
    RoomID      string
    Metadata    string // JSON
}

type CharacterDowntimeRepository struct {
    db *sql.DB
}

func NewCharacterDowntimeRepository(db *sql.DB) *CharacterDowntimeRepository {
    return &CharacterDowntimeRepository{db: db}
}

// Save upserts the downtime state for a character.
func (r *CharacterDowntimeRepository) Save(ctx context.Context, characterID int64, state DowntimeState) error {
    _, err := r.db.ExecContext(ctx, `
        INSERT INTO character_downtime (character_id, activity_id, completes_at, room_id, activity_metadata)
        VALUES ($1, $2, $3, $4, $5)
        ON CONFLICT (character_id) DO UPDATE SET
            activity_id       = EXCLUDED.activity_id,
            completes_at      = EXCLUDED.completes_at,
            room_id           = EXCLUDED.room_id,
            activity_metadata = EXCLUDED.activity_metadata`,
        characterID, state.ActivityID, state.CompletesAt, state.RoomID, state.Metadata)
    return err
}

// Load returns the active downtime state for a character, or nil if none.
func (r *CharacterDowntimeRepository) Load(ctx context.Context, characterID int64) (*DowntimeState, error) {
    row := r.db.QueryRowContext(ctx,
        `SELECT activity_id, completes_at, room_id, activity_metadata FROM character_downtime WHERE character_id = $1`,
        characterID)
    var s DowntimeState
    var meta sql.NullString
    err := row.Scan(&s.ActivityID, &s.CompletesAt, &s.RoomID, &meta)
    if err == sql.ErrNoRows {
        return nil, nil
    }
    if err != nil {
        return nil, err
    }
    s.Metadata = meta.String
    return &s, nil
}

// Clear removes the active downtime state for a character.
func (r *CharacterDowntimeRepository) Clear(ctx context.Context, characterID int64) error {
    _, err := r.db.ExecContext(ctx,
        `DELETE FROM character_downtime WHERE character_id = $1`, characterID)
    return err
}
```

```go
// TagsContain returns true if the room tags string contains the required tag.
// tags is a comma-separated string from room.Properties["tags"].
func TagsContain(tags, tag string) bool {
    for _, t := range splitTags(tags) {
        if t == tag {
            return true
        }
    }
    return false
}

func splitTags(s string) []string {
    // split on comma, trim spaces
    out := []string{}
    for _, part := range strings.Split(s, ",") {
        if trimmed := strings.TrimSpace(part); trimmed != "" {
            out = append(out, trimmed)
        }
    }
    return out
}

// AnalyzeTechTags returns the tags that satisfy the Analyze Tech location requirement
// (workshop OR archive, plus safe).
func AnalyzeTechTagsSatisfied(roomTags string) bool {
    return TagsContain(roomTags, "safe") &&
        (TagsContain(roomTags, "workshop") || TagsContain(roomTags, "archive"))
}
```

```go
// internal/game/downtime/engine.go
package downtime

import (
    "context"
    "time"
)

// DowntimeState is the subset of PlayerSession fields managed by the engine.
type DowntimeState struct {
    ActivityID     string
    CompletesAt    time.Time
    Busy           bool
    Metadata       string
}

// StartResult is returned by Start().
type StartResult struct {
    State       DowntimeState
    DurationMin int // actual duration chosen (relevant for craft/retrain)
}

// CanStart validates preconditions before starting an activity.
// Returns an error string if not allowed; empty string if OK.
// Precondition: roomTags is room.Properties["tags"]; currentBusy indicates an active activity.
func CanStart(activityAlias string, roomTags string, currentBusy bool) string {
    if currentBusy {
        return "You already have an active downtime activity." // REQ-DT-3
    }
    if !TagsContain(roomTags, "safe") {
        return "You must be in a Safe room to begin a downtime activity." // REQ-DT-1
    }
    act, ok := ActivityByAlias(activityAlias)
    if !ok {
        return "Unknown downtime activity."
    }
    // REQ-DT-2: validate specific tags
    if act.ID == "analyze_tech" {
        if !AnalyzeTechTagsSatisfied(roomTags) {
            return "Analyze Tech requires a room with a workshop or archive."
        }
        return "" // safe + (workshop|archive) satisfied
    }
    for _, tag := range act.RequiredTags {
        if tag == "safe" {
            continue // already checked above
        }
        if !TagsContain(roomTags, tag) {
            return "This room does not have the required facilities for that activity."
        }
    }
    return ""
}
```

```go
func (s *GameServiceServer) handleDowntime(uid string, req *gamev1.DowntimeRequest) error {
    sess, ok := s.sessions.GetPlayer(uid)
    if !ok {
        return fmt.Errorf("session not found")
    }

    sub := strings.ToLower(req.GetSubcommand())

    // Query current status
    if sub == "" {
        if !sess.DowntimeBusy {
            return s.sendConsole(uid, "No active downtime activity.")
        }
        act, _ := downtime.ActivityByID(sess.DowntimeActivityID)
        remaining := time.Until(sess.DowntimeCompletesAt).Round(time.Second)
        return s.sendConsole(uid, fmt.Sprintf("Downtime: %s — %s remaining.", act.Name, remaining))
    }

    // List activities
    if sub == "list" {
        var sb strings.Builder
        sb.WriteString("--- Downtime Activities ---\n")
        for _, a := range downtime.AllActivities() {
            tags := strings.Join(a.RequiredTags, "+")
            sb.WriteString(fmt.Sprintf("  %-12s %-28s %dmin  [%s]\n", a.Alias, a.Name, a.DurationMinutes, tags))
        }
        return s.sendConsole(uid, sb.String())
    }

    // Cancel
    if sub == "cancel" {
        if !sess.DowntimeBusy {
            return s.sendConsole(uid, "No active downtime activity to cancel.")
        }
        sess.DowntimeBusy = false
        sess.DowntimeActivityID = ""
        sess.DowntimeCompletesAt = time.Time{}
        sess.DowntimeMetadata = ""
        if s.downtimeRepo != nil {
            _ = s.downtimeRepo.Clear(context.Background(), sess.CharacterID)
        }
        return s.sendConsole(uid, "Downtime activity cancelled.") // REQ-DT-4: no material refund
    }

    // Start activity
    room, roomOK := s.world.GetRoom(sess.RoomID)
    if !roomOK {
        return s.sendError(uid, "Cannot determine room.")
    }
    roomTags := room.Properties["tags"]

    if errMsg := downtime.CanStart(sub, roomTags, sess.DowntimeBusy); errMsg != "" {
        return s.sendError(uid, errMsg)
    }

    act, _ := downtime.ActivityByAlias(sub)
    durationMin, metadata, err := s.downtimePreStart(uid, sess, act, req.GetArgs())
    if err != nil {
        return s.sendError(uid, err.Error())
    }

    now := time.Now() // use game clock time in production — replace with s.clock.Now()
    completesAt := now.Add(time.Duration(durationMin) * time.Minute)

    sess.DowntimeActivityID = act.ID
    sess.DowntimeCompletesAt = completesAt
    sess.DowntimeBusy = true
    sess.DowntimeMetadata = metadata

    if s.downtimeRepo != nil {
        state := postgres.DowntimeState{
            ActivityID:  act.ID,
            CompletesAt: completesAt,
            RoomID:      sess.RoomID,
            Metadata:    metadata,
        }
        _ = s.downtimeRepo.Save(context.Background(), sess.CharacterID, state)
    }

    return s.sendConsole(uid, fmt.Sprintf("You begin %s. Estimated completion: %s.", act.Name, completesAt.Format("15:04:05")))
}
```

`downtimePreStart` handles per-activity pre-flight (craft: validate recipe/deduct materials; retrain: roll duration; others: basic validation). Returns `(durationMinutes, metadata JSON, error)`.
